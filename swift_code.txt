public func adjustFontSize(_ buttons: [(String, Float)], _ targetFontSize: Float) -> String? {
    let dynamicTypeFontSizes: [Float] = [11.0, 13.0, 15.0, 17.0, 20.0, 24.0, 28.0, 34.0, 40.0, 46.0, 52.0]
    var closestFontSize: Float? = nil

    for size in dynamicTypeFontSizes {
        if size >= targetFontSize {
            closestFontSize = size
            break
        }
    }

    var matchingTitle: String? = nil

    for button in buttons {
        if closestFontSize == button.1 {
            matchingTitle = button.0
            break
        }
    }

    return matchingTitle
}
